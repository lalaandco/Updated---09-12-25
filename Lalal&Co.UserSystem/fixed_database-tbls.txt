-- Create orders table
CREATE TABLE IF NOT EXISTS orders (
    order_id INT(11) AUTO_INCREMENT PRIMARY KEY,
    user_email VARCHAR(255) NOT NULL,
    full_name VARCHAR(255) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    address TEXT NOT NULL,
    city VARCHAR(100) NOT NULL,
    postal_code VARCHAR(20) NOT NULL,
    payment_method ENUM('gcash','mastercard','paypal','cod') NOT NULL,
    total_amount DECIMAL(10,2) NOT NULL,
    order_notes TEXT DEFAULT NULL,
    order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    status ENUM('pending','confirmed','processing','shipped','delivered','cancelled') NOT NULL DEFAULT 'pending',
    tracking_number VARCHAR(100) DEFAULT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);


-- Create order_items table
CREATE TABLE IF NOT EXISTS order_items (
    item_id INT(11) AUTO_INCREMENT PRIMARY KEY,
    order_id INT(11) NOT NULL,
    product_id INT(11) NOT NULL,
    product_name VARCHAR(255) NOT NULL,
    product_price DECIMAL(10,2) NOT NULL,
    quantity INT(11) NOT NULL,
    subtotal DECIMAL(10,2) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_order FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    CONSTRAINT fk_product FOREIGN KEY (product_id) REFERENCES product_tbl(product_id) ON DELETE CASCADE
);

-- Create inventory transactions table to track all stock movements
CREATE TABLE IF NOT EXISTS inventory_transactions (
    transaction_id INT(11) AUTO_INCREMENT PRIMARY KEY,
    product_id INT(11) NOT NULL,
    transaction_type ENUM('restock', 'move_to_display', 'customer_purchase', 'return', 'adjustment') NOT NULL,
    quantity INT(11) NOT NULL COMMENT 'Positive for additions, negative for removals',

-- Create index for faster queries

-- Sales tracking table (automatically populated when orders are completed)
CREATE TABLE IF NOT EXISTS sales_records (
    sale_id INT(11) AUTO_INCREMENT PRIMARY KEY,
    order_id INT(11) NOT NULL,
    user_email VARCHAR(255) NOT NULL,
    total_amount DECIMAL(10,2) NOT NULL,
    payment_method ENUM('gcash','mastercard','paypal','cod') NOT NULL,
    sale_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fiscal_year INT(4) NOT NULL,
    fiscal_month INT(2) NOT NULL,
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    INDEX idx_sale_date (sale_date),
    INDEX idx_fiscal (fiscal_year, fiscal_month)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


-- Trigger to automatically create sale record when order is delivered
DELIMITER $$
CREATE TRIGGER after_order_delivered
AFTER UPDATE ON orders
FOR EACH ROW
BEGIN
    IF NEW.status = 'delivered' AND OLD.status != 'delivered' THEN
        INSERT INTO sales_records (order_id, user_email, total_amount, payment_method, sale_date, fiscal_year, fiscal_month)
        VALUES (
            NEW.order_id, 
            NEW.user_email, 
            NEW.total_amount, 
            NEW.payment_method,
            NOW(),
            YEAR(NOW()),
            MONTH(NOW())
        )
        ON DUPLICATE KEY UPDATE sale_id = sale_id; -- Prevent duplicates
    END IF;
END$$
DELIMITER ;

-- Create view for quick sales summary
CREATE OR REPLACE VIEW v_sales_summary AS
SELECT 
    DATE(sale_date) as date,
    COUNT(sale_id) as total_sales,
    SUM(total_amount) as revenue,
    AVG(total_amount) as avg_sale,
    fiscal_year,
    fiscal_month
FROM sales_records
GROUP BY DATE(sale_date), fiscal_year, fiscal_month;


-- Create view for inventory valuation
CREATE OR REPLACE VIEW v_inventory_value AS
SELECT 
    p.product_id,
    p.product_name,
    p.category,
    p.inventory_quantity,
    p.display_quantity,
    (p.inventory_quantity + p.display_quantity) as total_stock,
    p.product_price,
    (p.inventory_quantity + p.display_quantity) * p.product_price as total_value
FROM product_tbl p;


-- Create view for low stock alerts
CREATE OR REPLACE VIEW v_low_stock_alerts AS
SELECT 
    product_id,
    product_name,
    category,
    inventory_quantity,
    display_quantity,
    (inventory_quantity + display_quantity) as total_stock,
    CASE 
        WHEN display_quantity = 0 THEN 'OUT_OF_STOCK'
        WHEN display_quantity < 10 THEN 'CRITICAL'
        WHEN display_quantity < 20 THEN 'LOW'
        ELSE 'NORMAL'
    END as stock_status
FROM product_tbl
WHERE display_quantity < 20 OR inventory_quantity < 30
ORDER BY total_stock ASC;

DELIMITER $$
CREATE TRIGGER after_rating_insert
AFTER INSERT ON product_ratings
FOR EACH ROW
BEGIN
    DECLARE avg_rating DECIMAL(3,2);
    DECLARE total_count INT;
    DECLARE star1, star2, star3, star4, star5 INT;
    
    -- Calculate new statistics
    SELECT 
        ROUND(AVG(rating), 2),
        COUNT(*),
        SUM(CASE WHEN rating = 1 THEN 1 ELSE 0 END),
        SUM(CASE WHEN rating = 2 THEN 1 ELSE 0 END),
        SUM(CASE WHEN rating = 3 THEN 1 ELSE 0 END),
        SUM(CASE WHEN rating = 4 THEN 1 ELSE 0 END),
        SUM(CASE WHEN rating = 5 THEN 1 ELSE 0 END)
    INTO avg_rating, total_count, star1, star2, star3, star4, star5
    FROM product_ratings
    WHERE product_id = NEW.product_id;
    
    -- Update product table
    UPDATE product_tbl
    SET 
        average_rating = avg_rating,
        total_ratings = total_count,
        rating_1_star = star1,
        rating_2_star = star2,
        rating_3_star = star3,
        rating_4_star = star4,
        rating_5_star = star5
    WHERE product_id = NEW.product_id;
END$$

-- Create the fixed trigger
DELIMITER $$

CREATE TRIGGER deduct_warehouse_on_shipped
AFTER UPDATE ON orders
FOR EACH ROW
BEGIN
    -- Trigger fires when status changes TO 'shipped' from ANY other status
    -- Works for: confirmed→shipped, processing→shipped, confirmed→processing→shipped
    
    IF NEW.status = 'shipped' AND OLD.status != 'shipped' THEN
        
        -- Log warehouse deduction for each order item
        INSERT INTO inventory_transactions (
            product_id, 
            transaction_type, 
            quantity, 
            inventory_before, 
            inventory_after, 
            display_before, 
            display_after, 
            notes
        )
        SELECT 
            oi.product_id,
            'customer_purchase',
            -oi.quantity,
            p.inventory_quantity,
            p.inventory_quantity - oi.quantity,
            p.display_quantity,
            p.display_quantity,
            CONCAT('Order #', NEW.order_id, ' shipped - Warehouse stock deducted (from status: ', OLD.status, ')')
        FROM order_items oi
        INNER JOIN product_tbl p ON oi.product_id = p.product_id
        WHERE oi.order_id = NEW.order_id;
        
        -- Deduct warehouse inventory
        UPDATE product_tbl p
        INNER JOIN order_items oi ON p.product_id = oi.product_id
        SET p.inventory_quantity = p.inventory_quantity - oi.quantity
        WHERE oi.order_id = NEW.order_id;
        
    END IF;
END$$

    IN days_old INT

        notification_type,
        title,
        message,
        related_id,
        related_table,
        severity,
        triggered_by,

    DELETE FROM admin_notifications
    WHERE is_read = 1 
      AND read_at < DATE_SUB(NOW(), INTERVAL days_old DAY);

    nl.related_table,
    nl.severity,
    nl.triggered_by,

-- Create user_notifications table for persistent user notifications
CREATE TABLE IF NOT EXISTS user_notifications (
    notification_id INT(11) AUTO_INCREMENT PRIMARY KEY,
    user_email VARCHAR(255) NOT NULL,
    notification_type ENUM('order_confirmed', 'payment_verified', 'order_processing', 
                          'order_shipped', 'order_delivered', 'order_cancelled', 
                          'cancellation_approved', 'cancellation_rejected',
                          'refund_approved', 'refund_rejected') NOT NULL,
    title VARCHAR(255) NOT NULL,
    message TEXT NOT NULL,
    order_id INT(11) NOT NULL,
    severity ENUM('info', 'success', 'warning', 'error') DEFAULT 'info',
    is_read TINYINT(1) DEFAULT 0,
    link VARCHAR(500) DEFAULT NULL,
    icon VARCHAR(50) DEFAULT 'bx-bell',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    read_at TIMESTAMP NULL DEFAULT NULL,
    INDEX idx_user (user_email),
    INDEX idx_read (is_read),
    INDEX idx_created (created_at),
    INDEX idx_order (order_id),
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    FOREIGN KEY (user_email) REFERENCES users(email) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Create user_notification_log for tracking
CREATE TABLE IF NOT EXISTS user_notification_log (
    log_id INT(11) AUTO_INCREMENT PRIMARY KEY,
    notification_id INT(11) DEFAULT NULL,
    user_email VARCHAR(255) NOT NULL,
    notification_type VARCHAR(100) NOT NULL,
    title VARCHAR(255) NOT NULL,
    message TEXT NOT NULL,
    order_id INT(11) DEFAULT NULL,
    severity VARCHAR(50) DEFAULT NULL,
    action_taken VARCHAR(255) DEFAULT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    archived_at TIMESTAMP NULL DEFAULT NULL,
    INDEX idx_user (user_email),
    INDEX idx_created (created_at),
    INDEX idx_order (order_id),
    FOREIGN KEY (notification_id) REFERENCES user_notifications(notification_id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- Archive old read notifications
DELIMITER $$

CREATE PROCEDURE archive_user_notifications(IN days_old INT)
BEGIN
    INSERT INTO user_notification_log (
        notification_id, user_email, notification_type, title, message, order_id, severity, action_taken, created_at, archived_at
    )
    SELECT 
        notification_id, user_email, notification_type, title, message, order_id, severity,
        'auto_archived', created_at, NOW()
    FROM user_notifications
    WHERE is_read = 1 AND read_at < DATE_SUB(NOW(), INTERVAL days_old DAY);
    
    DELETE FROM user_notifications
    WHERE is_read = 1 AND read_at < DATE_SUB(NOW(), INTERVAL days_old DAY);
END$$

-- Unread notifications count per user
CREATE OR REPLACE VIEW v_user_unread_count AS
SELECT 
    user_email,
    COUNT(*) as unread_count,
    SUM(CASE WHEN severity = 'success' THEN 1 ELSE 0 END) as success_count,
    SUM(CASE WHEN severity = 'info' THEN 1 ELSE 0 END) as info_count,
    SUM(CASE WHEN severity = 'warning' THEN 1 ELSE 0 END) as warning_count,
    SUM(CASE WHEN severity = 'error' THEN 1 ELSE 0 END) as error_count,
    MIN(created_at) as oldest_notification,
    MAX(created_at) as newest_notification
FROM user_notifications
WHERE is_read = 0
GROUP BY user_email;

-- Recent user notification activity
CREATE OR REPLACE VIEW v_user_notification_activity AS
SELECT 
    un.notification_id,
    un.user_email,
    un.notification_type,
    un.title,
    un.message,
    un.order_id,
    un.severity,
    un.is_read,
    un.created_at,
    u.name as user_name,
    o.order_date,
    o.status as order_status
FROM user_notifications un
LEFT JOIN users u ON un.user_email = u.email
LEFT JOIN orders o ON un.order_id = o.order_id
ORDER BY un.created_at DESC;
